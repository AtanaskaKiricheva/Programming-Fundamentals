using System;
using System.Collections.Generic;
using System.Linq;
using System.Text.RegularExpressions;

public class Example
{
    public static void Main()
    {
        int affected = int.Parse(Console.ReadLine());
        int key = int.Parse(Console.ReadLine());
        double loss = 0;

        for (int i = 0; i < affected; i++)
        {
            string[] websites = Console.ReadLine().Split(' ');
            string name = websites[0];
            long visits = long.Parse( websites[1]);
            double commercial = double.Parse(websites[2]);

             loss += visits * commercial;

            Console.WriteLine(name);

        }

        Console.WriteLine($"Total Loss: {loss:f20}");
        Console.WriteLine($"Security Token: {Math.Pow(key,affected)}");








    }
}
