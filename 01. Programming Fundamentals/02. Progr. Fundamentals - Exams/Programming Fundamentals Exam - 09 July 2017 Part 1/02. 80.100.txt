using System;
using System.Collections.Generic;
using System.Linq;
using System.Text.RegularExpressions;

public class Example
{
    public static void Main()
    {
        List<int> distance = Console.ReadLine().Split(' ').Select(int.Parse).ToList();

        var sum = 0;
        bool isOut = false;

        while (distance.Count > 0)
        {
            int index = int.Parse(Console.ReadLine());

            if (index < 0)
            {
                index = 0;
                distance[0] = distance.Last();
                isOut = true;
            }
            else if (index > distance.Count - 1)
            {
                index = distance.Count - 1;
                distance[distance.Count - 1] = distance[0];
                isOut = true;
            }

            int removed = distance[index];
            sum += distance[index];
            if (isOut == false)
            {
                distance.RemoveAt(index);
            }


            for (int j = 0; j < distance.Count; j++)
            {
                if (distance[j] <= removed)
                {
                    distance[j] += removed;
                }
                else
                {
                    distance[j] -= removed;
                }

            }
            isOut = false;

        }
        Console.WriteLine(sum);





    }
}
