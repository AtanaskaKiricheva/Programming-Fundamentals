using System;
using System.Collections.Generic;
using System.Linq;
using System.Text.RegularExpressions;

public class Example
{
    public static void Main()
    {
        string text = Console.ReadLine();
        List<char> toedit = new List<char>();

        foreach (var item in text)
        {
            toedit.Add(item);
        }

        string Didimon = @"^([^a-zA-Z\-]+)";
        string Bojomon = @"^([a-zA-Z]+\-[a-zA-Z]+)";
        string invalid = @"^([\-a-zA-Z]+)";

        while (toedit.Count>0)
        {
            if (Regex.IsMatch(text,Didimon))
            {
                Console.WriteLine(Regex.Match(text, Didimon).Groups[1]);
                toedit.RemoveRange(0, Regex.Match(text, Didimon).Groups[1].Length);
                text = (string.Join("", toedit));
            }
            if (Regex.IsMatch(text,Bojomon))
            {
                Console.WriteLine(Regex.Match(text, Bojomon).Groups[1]);
                toedit.RemoveRange(0, Regex.Match(text, Bojomon).Groups[1].Length);
                text = (string.Join("", toedit));
            }
            if (Regex.IsMatch(text, invalid))
            {
                toedit.RemoveRange(0, Regex.Match(text, invalid).Groups[1].Length);
                text = (string.Join("", toedit));
             
            }
        }
    }
}
